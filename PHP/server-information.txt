$_SERVER is an array that contains information about the server and execution environment. The keys available include:

	PHP_SELF
	SERVER_ADDR
	SERVER_SOFTWARE
	REQUEST_METHOD
	REQUEST_TIME_FLOAT
	DOCUMENT_ROOT
	HTTP_ACCEPT_CHARSET
	HTTP_ACCEPT_LANGUAGE
	HTTP_HOST
	HTTP_USER_AGENT
	REMOTE_ADDR
	REMOTE_PORT
	REDIRECT_REMOTE_USER
	SERVER_ADMIN
	SERVER_SIGNATURE
	SCRIPT_NAME
	PHP_AUTH_DIGEST
	PHP_AUTH_PW
	PATH_INFO
	GATEWAY_INTERFACE
	SERVER_NAME
	SERVER_PROTOCOL
	REQUEST_TIME
	QUERY_STRING
	HTTP_ACCEPT
	HTTP_ACCEPT_ENCODING
	HTTP_CONNECTION
	HTTP_REFERER
	HTTPS
	REMOTE_HOST
	REMOTE_USER
	SCRIPT_FILENAME
	SERVER_PORT
	PATH_TRANSLATED
	REQUEST_URI
	PHP_AUTH_USER
	AUTH_TYPE
	ORIG_PATH_INFO
